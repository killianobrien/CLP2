\begin{align*} S'(x) &= 1 +2x + 3x^2 + \cdots + n x^{n-1}\\ &= \diff{}{x} \left[\frac{x^{n+1}-1}{x-1}\right] \qquad\qquad\qquad\qquad \text{use the quotient rule}\\ &= \frac{(x-1)\cdot (n+1)x^n - (x^{n+1}-1)\cdot 1}{(x-1)^2} \qquad \text{now clean it up}\\ &= \frac{nx^{n+1}-(n+1)x^n+1}{(x-1)^2}. \end{align*}

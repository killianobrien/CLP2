\begin{align*} \log\left[\left(1+\frac1{2x}\right)^{x}\right]&=x\log\left(1+\frac1{2x}\right)=\frac{1}{2y}\log(1+y)\\ &=\frac{1}{2y}\sum_{n=1}^\infty \frac{(-1)^{n+1}}{n}y^{n}\\ &=\frac{1}{2y}\left[y-\frac{y^2}{2}+\frac{y^3}{3}-\frac{y^4}{4}+\cdots\right]\\ &=\left[\frac{1}{2}-\frac{y}{4}+\frac{y^2}{6}-\frac{y^3}{8}+\cdots\right]\\ \lim_{x \to \infty}\log\left[\left(1+\frac1{2x}\right)^{x}\right]&=\lim_{y \to 0^+}\left[\frac{1}{2}-\frac{y}{4}+\frac{y^2}{6}-\frac{y^3}{8}+\cdots\right]=\frac{1}{2}\\ \lim_{x \to \infty}\left[\left(1+\frac1{2x}\right)^{x}\right]&=e^{1/2}=\sqrt{e} \end{align*}

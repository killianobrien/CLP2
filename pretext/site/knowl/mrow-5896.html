\begin{align*} \sum_{n=0}^\infty \frac{(-1)^n}{(2n+1) 3^n} &=\sqrt{3} \sum_{n=0}^\infty \frac{(-1)^n}{(2n+1)\big(\sqrt{3}\big)^{2n+1} }\\ & =\sqrt{3} \sum_{n=0}^\infty (-1)^n\frac{x^{2n+1}}{2n+1} \bigg|_{x=\frac{1}{\sqrt{3}}} =\sqrt{3}\,\arctan\frac{1}{\sqrt{3}}\\ &=\sqrt{3}\,\frac{\pi}{6} =\frac{\pi}{2\sqrt{3}} \end{align*}

Write

\begin{align*} \sum_{n=1}^\infty\frac{n+2}{n!}e^n &=\sum_{n=1}^\infty\frac{n}{n!}e^n + \sum_{n=1}^\infty\frac{2}{n!}e^n\\ &=\sum_{n=1}^\infty\frac{e^n}{(n-1)!} + 2\sum_{n=1}^\infty\frac{e^n}{n!}\\ &=e\sum_{n=1}^\infty\frac{e^{n-1}}{(n-1)!} + 2\sum_{n=1}^\infty\frac{e^n}{n!}\\ &=e\sum_{n=0}^\infty\frac{e^n}{n!} + 2\sum_{n=1}^\infty\frac{e^n}{n!} \end{align*}

Recall that \(e^x = \displaystyle\sum\limits_{n=0}^\infty\frac{x^n}{n!}\text{.}\) So

\begin{align*} \sum_{n=1}^\infty\frac{n+2}{n!}e^n &=e\Big[\sum_{n=0}^\infty\frac{x^n}{n!}\Big]_{x=e} +2\Big[\sum_{n=1}^\infty\frac{x^n}{n!}\Big]_{x=e}\\ & =e\Big[e^x\Big]_{x=e} + 2\Big[e^x-1\Big]_{x=e} = e^{e+1} + 2(e^e-1)\\ &=(e+2)e^e-2 \end{align*}

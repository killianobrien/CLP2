\begin{align*} \int \tan^3 x \sec^{-0.7}x \dee{x}&= \int \dfrac{\tan^3 x}{\sec^{0.7 x}} \dee{x} = \int \dfrac{\tan^3 x}{\sec^{1.7 x}}\sec x \dee{x}\\ &=\int \frac{\tan^2x}{\sec^{1.7}x}\cdot \sec x \tan x \dee{x}\\ &=\int \frac{u^2-1}{u^{1.7}} \dee{u} = \int \left(u^{0.3}-u^{-1.7}\right) \dee{u}\\ &=\frac{u^{1.3}}{1.3} + \frac{1}{0.7u^{0.7}}+C\\ &=\frac{1}{1.3}\sec^{1.3}x + \frac{1}{0.7\sec^{0.7}x}+C\\ &=\frac{1}{1.3}\sec^{1.3}x + \frac{1}{0.7}\cos^{0.7}x+C \end{align*}

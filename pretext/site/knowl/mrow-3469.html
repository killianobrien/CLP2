\begin{align*} \amp\int_3^\infty\frac{1}{(x-1)^2(x-2)}\,\dee{x}\\ &\hskip0.5in=\lim_{M\rightarrow\infty} \int_3^M\Big(-\frac{1}{(x-1)^2}-\frac{1}{x-1}+\frac{1}{x-2}\Big)\,\dee{x}\\ &\hskip0.5in=\lim_{M\rightarrow\infty} \Big[\frac{1}{x-1}-\log(x-1)+\log(x-2)\Big]_3^M\\ &\hskip0.5in=\lim_{M\rightarrow\infty} \Big[\frac{1}{M-1}+\log\frac{M-2}{M-1}\Big] -\Big[\frac{1}{3-1}+\log\frac{3-2}{3-1}\Big]\\ &\hskip0.5in=\log 2-\frac{1}{2}\approx0.193 \end{align*}

\begin{align*} \hskip-8mm s_N &= \sum_{n=3}^N \bigg( \!\cos\Big( \frac \pi n \Big) - \cos\Big( \frac \pi{n+1} \Big) \bigg)\\ &=\cos\Big( \frac \pi 3 \Big) - \cos\Big( \frac \pi{N+1} \Big)\\ &=\frac12 - \cos\Big( \frac \pi{N+1} \Big).\\ \end{align*}

\begin{align*} \diff{}{x} \left[ x\sqrt{x^2-1} \right] &=\left[ \sqrt{x^2-1} + \frac{x^2}{\sqrt{x^2-1}} \right]\\ &= \frac{1}{\sqrt{x^2-1}} \left[(x^2-1) + x^2 \right]\\ \diff{}{x} \log\bigg| x +\sqrt{x^2-1} \bigg| &= \frac{1}{x+\sqrt{x^2-1}} \cdot \left[ 1+\frac{x}{\sqrt{x^2-1}} \right]\\ &= \frac{1}{x+\sqrt{x^2-1}} \cdot \frac{x+\sqrt{x^2-1}}{\sqrt{x^2-1}}\\ &= \frac{1}{\sqrt{x^2-1}} \end{align*}

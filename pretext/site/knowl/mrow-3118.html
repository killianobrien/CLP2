\begin{align*} \int_e^\infty\frac{\dee{x}}{x(\log x)^p} &=\lim_{R\rightarrow \infty} \int_e^R\frac{\dee{x}}{x(\log x)^p} \qquad\qquad\qquad \text{use substitution}\\ &=\lim_{R\rightarrow \infty} \int_1^{\log R}\frac{\dee{u}}{u^p} \qquad\qquad\text{with }u=\log x,\dee{u}=\frac{\dee{x}}{x}\\ &=\lim_{R\rightarrow\infty} \begin{cases} \frac{1}{1-p}\Big[(\log R)^{1-p}-1\Big] & \text{if } p\ne 1\\ \log(\log R) & \text{if } p=1 \end{cases}\\ &=\begin{cases} \text{divergent} & \text {if } p\le 1\\ \frac{1}{p-1} & \text{if } p \gt 1 \end{cases} \end{align*}

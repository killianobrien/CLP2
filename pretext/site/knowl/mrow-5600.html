\begin{align*} \frac{1}{1-x}&=\sum_{n=0}^\infty x^n \qquad \text{when }\\ \diff{}{x}\left\{\frac{1}{1-x}\right\}&=\diff{}{x}\left\{\sum_{n=0}^\infty x^n\right\}\\ \frac{1}{(1-x)^2}&=\sum_{n=1}^\infty nx^{n-1}\\ &= 1x^0 + 2x^1+\sum_{n=3}^\infty nx^{n-1}\\ &= 1 + 2x+\sum_{n=3}^\infty nx^{n-1}\\ \text{So,}\qquad \frac{1}{(1-x)^2}-1-2x&=\sum_{n=3}^\infty nx^{n-1}\\ \text{Setting }\qquad \frac{1}{(1-1/5)^2}-1-\frac25&=\sum_{n=3}^\infty n\left(\frac15\right)^{n-1}\\ \left(\frac{5}{4} \right)^2-1-\frac25&=\sum_{n=3}^\infty \frac{n}{5^{n-1}} \end{align*}

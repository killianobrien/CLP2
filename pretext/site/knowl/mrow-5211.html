\begin{align*} \frac{a_{k+1}}{a_k}&=\frac{\frac{e^{k+1}}{(k+1)!}}{\frac{e^k}{k!}}=\frac{e^{k+1}}{e^k}\cdot\frac{k!}{(k+1)!}\\ \amp= e\cdot\frac{k(k-1)\cdots(1)}{(k+1)(k)(k-1)\cdots(1)}=e\cdot\frac{1}{k+1}=\frac{e}{k+1}\\ \end{align*}

Since \(e\) is a constant,

\begin{align*} \lim_{k \rightarrow \infty}\frac{a_{k+1}}{a_k}&=\lim_{k \rightarrow \infty}\frac{e}{k+1}=0 \end{align*}

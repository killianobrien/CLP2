\begin{align*} \mbox{Area}&=16\int_{0}^{\pi/6} \sqrt{1-\left(\sin\theta\right)^2} \cdot 2\cos\theta\ \dee{\theta}\\ &=32\int_{0}^{\pi/6} \sqrt{\cos^2\theta} \cos\theta\ \dee{\theta}\\ &=32\int_{0}^{\pi/6}\cos^2\theta\ \dee{\theta}\\ &=16\int_{0}^{\pi/6}\big(1+\cos(2\theta)\big)\ \dee{\theta}\\ &=16\left[\theta +\frac{1}{2}\sin(2\theta)\right]_{0}^{\pi/6}\\ &=16\left(\frac{\pi}{6}+\frac{1}{2}\cdot \frac{\sqrt{3}}{2} \right)\\ &=\frac{8\pi}{3}+4\sqrt{3} \end{align*}

\begin{align*} \sum_{n=5}^\infty \frac{1}{e^n}&=\sum_{n=5}^\infty \left(\frac{1}{e}\right)^n\\ &=\sum_{n=0}^\infty \left(\frac{1}{e}\right)^n-\sum_{n=0}^4 \left(\frac{1}{e}\right)^n\\ &=\dfrac{1}{1-\frac{1}{e}}-\dfrac{1-\left(\frac{1}{e}\right)^{5}}{1-\frac{1}{e}}\\ &=\dfrac{\left(\frac{1}{e}\right)^5}{1-\frac{1}{e}} =\dfrac{1}{e^5\left(1-\frac{1}{e}\right)}\\ &=\dfrac{1}{e^5-e^4} \end{align*}

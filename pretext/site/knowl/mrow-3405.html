\begin{align*} \amp\int_3^\infty\frac{x}{(x-1)^2(x-2)}\,\dee{x}\\ \amp\hskip0.25in=\lim_{M\rightarrow\infty} \int_3^M\left(-\frac{1}{(x-1)^2}-\frac{2}{x-1}+\frac{2}{x-2}\right)\,\dee{x}\\ &\hskip0.25in=\lim_{M\rightarrow\infty} \left[\frac{1}{x-1}-2\log|x-1|+2\log|x-2|\right]_3^M\\ &\hskip0.25in=\lim_{M\rightarrow\infty} \bigg[\frac{1}{x-1}+2\log\bigg|\frac{x-2}{x-1}\bigg|\bigg]_3^M\\ &\hskip0.25in=\lim_{M\rightarrow\infty} \bigg[\frac{1}{M-1}+2\log\bigg|\frac{M-2}{M-1}\bigg|\bigg] -\bigg[\frac{1}{3-1}+2\log\bigg|\frac{3-2}{3-1}\bigg|\bigg]\\ &\hskip0.25in=2\log 2-\frac{1}{2}\approx0.886 \end{align*}

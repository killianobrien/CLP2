\begin{align*} \frac{\frac{1}{k}}{\frac{1}{\sqrt{k^2+k}}}&=\frac{\sqrt{k^2+k}}{k}=\frac{\sqrt{k^2+k}}{\sqrt{k^2}}= \sqrt{\frac{k^2+k}{k^2}}=\sqrt{1+\frac{1}{k}},\mbox{ so}\\ \lim_{k \rightarrow \infty}\frac{\frac{1}{k}}{\frac{1}{\sqrt{k^2+k}}}&=\lim_{k \rightarrow \infty}\sqrt{1+\frac{1}{k}}=1 \end{align*}

\begin{align*} \int 3x^2 \cos(x^3) \dee{x} &= \int u'(x) \cos\big(u(x)\big) \dee{x} &\text{just swap order of factors}\\ &= \int \cos\big(u(x)\big) u'(x) \dee{x} & \text{by the substitution rule}\\ &= \int \cos(u) \dee{u}\bigg|_{u=x^3}\\ &= \left( \sin(u) + C\right)\bigg|_{u=x^3} & \text{using Theorem }\knowl{./knowl/thm_imp_indef_int.html}{\text{1.3.17}})\\ &= \sin(x^3)+C \end{align*}

\begin{align*} \frac{1}{3}\int_0^1 \sin\big(x^3+1\big)\cdot 3x^2 \dee{x} &=\frac{1}{3}\int_0^1 \underbrace{\sin\big(x^3+1\big)}_{=f(u(x))}\cdot \underbrace{3x^2}_{=u'(x)} \dee{x}\\ \end{align*}

Now set \(u(x)=x^3+1\) and \(f(u)=\sin(u)\)

\begin{align*} &= \frac{1}{3}\int_0^1 f(u(x)) u'(x) \dee{x}\\ &= \frac{1}{3} \int_{u(0)}^{u(1)} f(u) \dee{u} & \text{by the substitution rule}\\ &= \frac{1}{3} \int_{1}^{2} \sin(u) \dee{u} & \text{since $u(0)=1$ and $u(1)=2$}\\ &= \frac{1}{3} \big[-\cos(u)\big]_1^2\\ &= \frac{1}{3}\big( -\cos(2) -(-\cos(1)) \big)\\ &= \frac{\cos(1)-\cos(2)}{3}. \end{align*}

\begin{align*} \int \tan^3 x\,\dee{x} &=\int \frac{\sin^2x}{\cos^3x}\sin x\,\dee{x} \qquad \text{substitute $u=\cos x$}\\ &=\int \frac{1-u^2}{u^3} (-1) \dee{u}\\ &=\frac{u^{-2}}{2}+\log|u|+C\\ &=\frac{1}{2 \cos^2 x}+\log\left|\cos x\right|+C \qquad \text{rewrite in terms of secant}\\ &= \frac{1}{2}\sec^2 x - \log\left|\sec x\right|+C \end{align*}
